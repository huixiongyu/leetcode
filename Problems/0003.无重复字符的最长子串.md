## C++

```
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int tmp[130] = {0}, len = s.size(), max = 0;
        for(int i = 0, j = 0; j < len; j++) {
            tmp[s[j]]++;
            // 如果出现重复的字母，开始清除之前的记录重新计算
            while(tmp[s[j]] > 1)
                tmp[s[i++]]--;
            if (j - i + 1 > max)
                max = j - i + 1;
        }
        return max;
    }
};
```
